    <script>
        const yesBtn = document.getElementById('yesBtn');
        const noBtn = document.getElementById('noBtn');
        const questionSection = document.getElementById('question-section');
        const successMessage = document.getElementById('success-message');

        // The specific messages you asked for
        const messages = [
            "are you sure?",
            "baby u sure",
            "babyyyyyy",
            "just say yes baby",
            "baby pleaseeee"
        ];

        let messageIndex = 0;
        let currentFontSize = 1.2; // Starting font size in rem

        noBtn.addEventListener('click', function() {
            // 1. Change the text of the No button
            noBtn.textContent = messages[messageIndex];
            
            // Cycle through the messages
            messageIndex = (messageIndex + 1) % messages.length;

            // 2. Make the Yes button bigger
            currentFontSize += 0.8; // Increase size significantly each click
            yesBtn.style.fontSize = `${currentFontSize}rem`;
            
            // Optional: increase padding slightly to match font size growth
            const currentPadding = 15 + (currentFontSize * 5);
            yesBtn.style.padding = `${currentPadding}px ${currentPadding * 2}px`;
        });

        yesBtn.addEventListener('click', function() {
            // Hide the question and show the love message
            questionSection.style.display = 'none';
            successMessage.style.display = 'block';
        });
    </script>
